定制 UIButton

2012-12-05 21:13

做到以下设定——

  • 大小和位置
  • 文字左对齐
  • 字体，字体大小和颜色
  • 文字旁插入小图
  • 边距
  • 边框颜色
  • 圆角弧度
  • 背景颜色

codes--

UIButton *button = [[UIButton alloc] initWithFrame:CGRectMake(0, 0, 457, 55)];
[button setTitle:@"Select A Practice Mode" forState:UIControlStateNormal];
button.center = CGPointMake(1024*0.5, 185);
button.backgroundColor = [UIColor blackColor];
button.layer.cornerRadius = 15;
button.layer.masksToBounds = YES;
button.titleLabel.font = [UIFont fontWithName:@"EuphemiaUCAS-Bold" size:22];
button.contentHorizontalAlignment = UIControlContentHorizontalAlignmentLeft;
button.contentEdgeInsets = UIEdgeInsetsMake(0, 70, 0, 0);
UIImage *btnImage = [UIImage imageNamed:@"setBtnIcon_0.png"];
button.imageEdgeInsets = UIEdgeInsetsMake(0, -20, 0, 0);
[button setImage:btnImage forState:UIControlStateNormal];
button.layer.borderWidth = 2.0f;
button.layer.borderColor = [UIColor colorWithRed:107.0 / 255 green:107.0 / 255 blue:107.0 / 255 alpha:1.0].CGColor;
button.tag = 201;
[button addTarget:self action:@selector(dismissAddFeed:) forControlEvents:UIControlEventTouchUpInside];
[self.view addSubview:button];
