定制 UIPickerView

2012-12-06 21:19

Picker 带的背景颜色，最简单的办法，用一张图片把它遮住

timePicker = [[UIPickerView alloc] initWithFrame:CGRectZero];
timePicker.frame = CGRectMake(131.0, 155.0, 764.0, 216.0);
timePicker.delegate = self;
timePicker.dataSource = self;
timePicker.showsSelectionIndicator = YES;
[detailView addSubview:timePicker];
NSUserDefaults *userDefaults = [NSUserDefaults standardUserDefaults];
int currentTime = 1;
currentTime = [userDefaults integerForKey:@"drillTime"];
[timePicker selectRow:(currentTime - 1) inComponent:0 animated:YES];

UIView *pickerBg = [[UIView alloc] initWithFrame:CGRectMake(0, 0, 764, 216)];
CALayer *bgbg = [CALayer layer];
bgbg.bounds = CGRectMake(0, 0, 764, 216);
bgbg.position = CGPointMake(764.0*0.5, 216.0*0.5);
bgbg.backgroundColor = [UIColor colorWithPatternImage:[UIImage imageNamed:@"timePickerMask.png"]].CGColor;
[pickerBg.layer addSublayer:bgbg];
//pickerBg.backgroundColor = [UIColor colorWithPatternImage:[UIImage imageNamed:@"bgPicker.png"]];
pickerBg.userInteractionEnabled = NO;
[timePicker insertSubview:pickerBg atIndex:100];
pickerBg = nil;
